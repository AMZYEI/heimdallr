# installation (Debian)
    $ cd debian
    $ ./preinstallRequirementsDebian.sh
    $ ./install-heimdallr.py


# installation (Fedora)
    $ cd fedora
    $ ./preinstallRequirementsFedora.sh
    $ ./install-heimdallr.py


# update
    $ heimdallr-update

# exec (local)
    $ cd debian OR cd fedora
    $ ./preinstallRequirements*.sh
    $ make
    $ ./heimdallr
